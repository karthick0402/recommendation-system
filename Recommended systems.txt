import numpy as np
import pandas as pd
 
dataset = pd.read_csv(r"movies.csv")

import re
import nltk
nltk.download("stopwords")
from nltk.corpus import stopwords
from nltk.stem.porter import PorterStemmer
ps = PorterStemmer()

genre = []
for i in range(0,9742):
    review = dataset["genres"][i]
    review = re.sub ('[^a-zA-Z]'," ",review)
    review = review.lower()
    review = review.split()
    review = [ps.stem(word) for word in review if not word in set(stopwords.words("english"))]
    review = " ".join(review)
    genre.append(review)
    
from sklearn.preprocessing import LabelEncoder
le = LabelEncoder()
dataset["title"] = le.fit_transform(dataset["title"])

from sklearn.feature_extraction.text import CountVectorizer
cv = CountVectorizer(max_features = 2000)
x = cv.fit_transform(genre).toarray()
y = dataset.iloc[:,1:2].values

from sklearn.preprocessing import OneHotEncoder
one = OneHotEncoder()
z = one.fit_transform(y[:,0:1]).toarray()
y = np.delete(y,0,axis =1)
y = np.concatenate((z,y),axis = 1)

from sklearn.model_selection import train_test_split
x_train,x_test,y_train,y_test = train_test_split(x,y,test_size = 0.2,random_state = 0)

from keras.models import Sequential
from keras.layers import Dense
model = Sequential()    

model.add(Dense(units = 2000 , init = "random_uniform", activation = "relu"))
model.add(Dense(units = 2500 , init = "random_uniform", activation = "relu"))
model.add(Dense(units = 2500 , init = "random_uniform", activation = "relu"))
model.add(Dense(units = 9737 , init = "random_uniform", activation = "softmax"))
model.compile(optimizer = "adam",loss= "categorical_crossentropy",metrices = ["accuracy"])
model.fit(x_train,y_train,epochs = 10,batch_size = 128)

y_pred = model.predict(x_test)
y_pred_model = model.predict_classes(x_test)
